题目大意

在一个火车站中，火车按顺序进站（从小到大），然后判断题目给的出站顺序是否可能出现。
很容易知道，火车站满足“后进先出”的特点，因此用栈来实现。

// Problem#: 1509
// Submission#: 3415538
// The source code is licensed under Creative Commons Attribution-NonCommercial-ShareAlike 3.0 Unported License
// URI: http://creativecommons.org/licenses/by-nc-sa/3.0/
// All Copyright reserved by Informatic Lab of Sun Yat-sen University

#include <cstdio>
#include <stack>
#include <iostream>
using namespace std;
int a[1001];
int main(){
    int n;
    while(cin >> n&&n){
        while(cin >> a[0]){
            stack<int> s;
            if(a[0]==0) {
                cout << endl;
                break;
            }

            for(int i=1;i<n;i++)  cin >> a[i];
            int j = 0;
            for(int i=1;i<=n;i++){
                s.push(i);
                while(!s.empty()){
                    if(s.top()==a[j]) 
                    {               
                        s.pop();
                        j++;
                    }
                    else break;
                }
            }

            if(s.empty()) cout << "Yes" <<endl;
            else cout << "No" <<endl;
        }
    }
    return 0;
}
